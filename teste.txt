-- Setando o Diretório:
cd OneDrive

-- Iniciando controle de versão de todo o diretório:
git init

-- Selecionando o arquivo text.txt para o controle de versões:
git add text.txt

-- Verificando o status do controle de versões:
git status

-- Salvando as mudanças feitas no teste.txt:
git add text.txt

-- Confirmando as mudanças feitas, com mensagem (first commit):
git commit -m "first commit"
